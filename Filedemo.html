    private void createActionPerformed(java.awt.event.ActionEvent evt) {                                       
        String path;
        path = input1.getText();
        File file = new File(path);
        if(!file.exists())
        {
            if(file.mkdirs())
            {
                output.setText("Directory Created...");
            }
            else
            {
                output.setText("Error Ceation !!");
            }
        }
    }                                      

    private void deleteActionPerformed(java.awt.event.ActionEvent evt) {                                       
        String path;
        path = input1.getText();
        File file = new File(path);
        if(file.delete())
        {
            output.setText("Folder Deleted..");
        }
        else
        {
            output.setText("Error Deleting !!");
        }
    }                                      

    private void renameActionPerformed(java.awt.event.ActionEvent evt) {                                       
        String path1;
        String path2;
        path1 = input1.getText();
        path2 = input2.getText();
        File oldfile = new File(path1);
        File newfile = new File(path2);
        boolean success = oldfile.renameTo(newfile);
        if(success)
        {
            output.setText("Folder Renamed..");
        }
        else
        {
           output.setText("Error Renaming !!");
        }
    }                                      

    private void textfileActionPerformed(java.awt.event.ActionEvent evt) {                                         
        String path;
        path = input1.getText();
        try
        {
            File file = new File(path);
            if(file.createNewFile())
            {
                output.setText("Text File Created..");
            }
            else
            {
                output.setText("Error Creating !!");
            }
        }
        catch(IOException e)
        {
        }
    }                                        

    private void openActionPerformed(java.awt.event.ActionEvent evt) {                                     
        String path;
        path = input1.getText();
        File file = new File(path);
        if(!Desktop.isDesktopSupported())
        {
            output.setText("Unsupported File !!");
        }
        Desktop desktop = Desktop.getDesktop();
        if(file.exists())
        {
            try {
                desktop.open(file);
            } 
            catch (IOException ex) 
            {
                Logger.getLogger(Filedemo.class.getName()).log(Level.SEVERE, null, ex);
            }
        }
    }                                    

    private void treeActionPerformed(java.awt.event.ActionEvent evt) {                                     
        String path;
        path = input1.getText();
        File file = new File(path);
        output2.setText("------------- FILES FROM DIRECTORY -------------\n");
        if(file.exists() && file.isDirectory())
        {
            File[]arr = file.listFiles();
            RecursivePrint(arr,0,0);
            }
        else
        {
            output2.setText("Error Finding File !!");
        }
    }                                    

    /**
     *
     * @param arr
     * @param index
     * @param level
     */
    public void RecursivePrint(File[] arr,int index, int level)
    {
        if(index == arr.length)
            return;
          
        // tabs for internal levels
        for (int i = 0; i < level; i++)
            output2.append("\t");
          
         // for files
        if(arr[index].isFile()) 
        {
            output2.append(arr[index].getName()+"\n");
        }
        //for sub - drectories
        else if(arr[index].isDirectory())
        {
            output2.append("[" +arr[index].getName() + "]\n");
            RecursivePrint(arr[index].listFiles(),0,level+1);
        }
        RecursivePrint(arr,++index, level);
        
       
    }
